# 素材路径

MODEL_DIR = "./素材/模特照片/"
OUTPUT_DIR = "./输出/春节模板/"

# 模特列表
MODELS = ["阿峰", "成俊", "小佳", "木凡"]

# 技能列表
SKILLS = {

    "Skills-2": "动漫脸高精度",
    "Skills-3": "真人+动漫同框"
}

def generate_image(skill_name, model_name, background_path, model_path):
    """生成单张图片"""
    # 读取图片
    with open(background_path, "rb") as f:
        background = base64.b64encode(f.read()).decode()
    
    with open(model_path, "rb") as f:
        model = base64.b64encode(f.read()).decode()
    
    # 构建请求
    payload = {
        "model": MODEL_NAME,
        "prompt": SKILLS_PROMPTS[skill_name],
        "image": [background, model],  # 待确认格式
        "n": 1,
        "size": "2048x2048",
        "response_format": "b64_json"
    }
    
    headers = {
        "Authorization": f"Bearer {API_KEY}",
        "Content-Type": "application/json"
    }
    
    # 发送请求
    response = requests.post(API_ENDPOINT, json=payload, headers=headers)
    
    if response.status_code == 200:
        result = response.json()
        # 保存图片
        output_name = f"{skill_name}_{model_name}_{datetime.now().strftime('%Y%m%d_%H%M%S')}.png"
        output_path = os.path.join(OUTPUT_DIR, output_name)
        
        image_data = base64.b64decode(result["data"][0]["b64_json"])
        with open(output_path, "wb") as f:
            f.write(image_data)
        
        return output_path
    else:
        print(f"生成失败: {response.text}")
        return None

def batch_generate():
    """批量生成所有组合"""
    for skill_name in SKILLS.keys():
        for model_name in MODELS:
            print(f"正在生成: {skill_name} - {model_name}")
            
            # 遍历所有背景模板
